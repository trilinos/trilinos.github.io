100template<typename int_type>
101typename TCrsGraph_MapColoringIndex<int_type>::NewTypeRef
102TCrsGraph_MapColoringIndex<int_type>::
103operator()( OriginalTypeRef orig )
104{
105 if(!orig.RowMap(). template GlobalIndicesIsType<int_type>())
106 throw "EpetraExt::TCrsGraph_MapColoringIndex::operator(): Global indices mismatch.";
107
108 Base::origObj_ = &orig;
109
110 const Epetra_BlockMap & RowMap = orig.RowMap();
111 int nRows = RowMap.NumMyElements();
112
113 int NumColors = ColorMap_.NumColors();
114 int * ListOfColors = ColorMap_.ListOfColors();
115
116 std::map<int,int> MapOfColors;
117 for( int i = 0; i < NumColors; ++i ) MapOfColors[ ListOfColors[i] ] = i;
118
119 //initial setup of stl vector of IntVectors for indexing
120 std::vector<int_type> dummy( nRows, -1 );
121 typename Base::NewTypePtr IndexVec = new typename Base::NewType( NumColors, typename Epetra_GIDTypeVector<int_type>::impl( Copy, RowMap, &dummy[0] ) );
122
123 int MaxNumIndices = orig.MaxNumIndices();
124 int NumIndices;
125 std::vector<int_type> Indices( MaxNumIndices );
126
127 for( int i = 0; i < nRows; ++i )
128 {
129 orig.ExtractGlobalRowCopy( (int_type) orig.GRID64(i), MaxNumIndices, NumIndices, &Indices[0] );
130
131 for( int j = 0; j < NumIndices; ++j )
132 (*IndexVec)[ MapOfColors[ColorMap_(Indices[j])] ][i] = Indices[j];
133 }
134
135 Base::newObj_ = IndexVec;
136
137 return *IndexVec;
138}
