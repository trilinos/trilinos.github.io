52 using MVT = MultiVecTraits<ScalarType, MV>;
53 return (isInitialized() &&
54 !R.is_null() &&
55 !Z.is_null() &&
56 !P.is_null() &&
57 !AP.is_null() &&

59 (MVT::GetGlobalLength(*tmp) == MVT::GetGlobalLength(*R)));
