242template<class ObjType>
243class ConstNonconstObjectContainer {
244public:
246 ConstNonconstObjectContainer()
247 :constObj_(null),isConst_(true) {}
249 ConstNonconstObjectContainer( const RCP<ObjType> &obj )
250 { initialize(obj); }
252 ConstNonconstObjectContainer( const RCP<const ObjType> &obj )
253 { initialize(obj); }
256 void initialize( const RCP<ObjType> &obj )
257 {
258 TEUCHOS_TEST_FOR_EXCEPTION(is_null(obj), NullReferenceError, "Error!");
259 constObj_ = obj;
260 isConst_ = false;
261 }
264 void initialize( const RCP<const ObjType> &obj )
265 {
266 TEUCHOS_TEST_FOR_EXCEPTION(is_null(obj), NullReferenceError, "Error!");
267 constObj_ = obj;
268 isConst_ = true;
269 }
271 void uninitialize()
272 { constObj_=null; isConst_=true; }
274 ConstNonconstObjectContainer<ObjType>& operator=(ENull)
275 { uninitialize(); return *this; }
277 bool isConst() const
278 { return isConst_; }
293 RCP<ObjType> getNonconstObj() const
294 {
295 TEUCHOS_TEST_FOR_EXCEPTION(
296 constObj_.get() && isConst_, NonconstAccessError,
297 "Error, the object of reference type \""<<TypeNameTraits<ObjType>::name()
298 <<"\" was given as a const-only object and non-const access is not allowed."
299 );
300 return rcp_const_cast<ObjType>(constObj_);
301 }
307 RCP<const ObjType> getConstObj() const
308 { return constObj_; }
310 RCP<const ObjType> operator()() const
311 { return getConstObj(); }
318 const ObjType* operator->() const
319 { return &*getConstObj(); } // Does assert also!
326 const ObjType& operator*() const
327 { return *getConstObj(); }
329 operator RCP<const ObjType>() const
330 { return getConstObj(); }
331
332private:
333 RCP<const ObjType> constObj_;
334 bool isConst_;
335};
