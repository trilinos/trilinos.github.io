19RCP<const ParameterEntryXMLConverter>
20ParameterEntryXMLConverterDB::getConverter(RCP<const ParameterEntry> entry)
21{
22 ConverterMap::const_iterator it =
23 getConverterMap().find(entry->getAny().typeName());
24 if(it == getConverterMap().end()){
25 return getDefaultConverter();
26 }
27 else{

32RCP<const ParameterEntryXMLConverter>
33ParameterEntryXMLConverterDB::getConverter(const XMLObject& xmlObject)
34{
35 TEUCHOS_TEST_FOR_EXCEPTION(
36 !xmlObject.hasAttribute(ParameterEntryXMLConverter::getTypeAttributeName()),
37 NoTypeAttributeExecption,
38 ParameterEntry::getTagName() <<" tags must "
39 "have a " << ParameterEntryXMLConverter::getTypeAttributeName() <<
40 " attribute." << std::endl <<
41 "Bad Parameter: " <<
42 xmlObject.getAttribute(XMLParameterListWriter::getNameAttributeName()) <<
43 std::endl << std::endl);
44
45 std::string parameterType = xmlObject.getRequired(
46 ParameterEntryXMLConverter::getTypeAttributeName());
47 ConverterMap::const_iterator it = getConverterMap().find(parameterType);
48
49 TEUCHOS_TEST_FOR_EXCEPTION(it == getConverterMap().end(),
50 CantFindParameterEntryConverterException,
51 "Can't find converter for parameter entry of type: " <<
52 xmlObject.getRequired(ParameterEntryXMLConverter::getTypeAttributeName()) <<
53 std::endl << std::endl);
54
55 return it->second;
56}
