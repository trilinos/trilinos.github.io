525/* \brief Find the permutation from the expected nodes to the actual nodes,
526 *
527 * Find permutation 'p' such that:
528 * actual_node[j] == expected_node[ top.permutation[p].node[j] ]
529 * for all vertices.
530 */
531template< typename id_type >
532int findPermutation( const CellTopologyData & top ,
533 const id_type * const expected_node ,
534 const id_type * const actual_node )
535{
536 const int nv = top.vertex_count ;
537 const int np = top.permutation_count ;
538 int p = 0 ;
539 for ( ; p < np ; ++p ) {
540 const unsigned * const perm_node = top.permutation[p].node ;
541 int j = 0 ;
542 for ( ; j < nv && actual_node[j] == expected_node[ perm_node[j] ] ; ++j );
543 if ( nv == j ) break ;
544 }
545 if ( np == p ) p = -1 ;
546 return p ;
547}
548
549template< typename id_type >
550int findPermutation( const CellTopology & top ,
551 const id_type * const expected_node ,
552 const id_type * const actual_node )
553{
554 return findPermutation( * top.getCellTopologyData() , expected_node , actual_node );
555}

582inline
583unsigned cellTopologyKey( const unsigned dimension ,
584 const unsigned face_count ,
585 const unsigned edge_count ,
586 const unsigned vertex_count ,
587 const unsigned node_count )
588{
589 const bool bad = ( dimension >> 3 ) ||
590 ( face_count >> 6 ) ||
591 ( edge_count >> 6 ) ||
592 ( vertex_count >> 6 ) ||
593 ( node_count >> 10 );
594
595 if ( bad ) {
596 badCellTopologyKey( dimension ,
597 face_count ,
598 edge_count ,
599 vertex_count ,
600 node_count );
601 }
602
603 const unsigned key = ( dimension << 28 ) |
604 ( face_count << 22 ) |
605 ( edge_count << 16 ) |
606 ( vertex_count << 10 ) |
607 ( node_count ) ;
608
609 return key ;
610}
