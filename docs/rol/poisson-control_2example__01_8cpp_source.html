48
51 virtual bool check( ROL::AlgorithmState<Real> &state ) {
52 if ( (state.gnorm > this->gtol_) &&
53 (state.snorm > this->stol_) &&
54 (state.iter < this->max_iter_) ) {
55 return true;
56 }
57 else {
58 if ( state.iter < 2 ) {
59 return true;
60 }
61 return false;
