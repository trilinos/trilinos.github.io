14template<class Real>
15Real random(const ROL::Ptr<const Teuchos::Comm<int> > &comm) {
16 Real val = 0.0;
17 if ( Teuchos::rank<int>(*comm)==0 ) {
18 val = (Real)rand()/(Real)RAND_MAX;
19 }
20 Teuchos::broadcast<int,Real>(*comm,0,1,&val);
21 return val;
22}
