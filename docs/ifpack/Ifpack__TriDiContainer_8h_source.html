199 virtual int SetNumVectors(const int NumVectors_in)
200 {
201 if (NumVectors_ == NumVectors_in)
202 return(0);
203
204 NumVectors_ = NumVectors_in;
205 IFPACK_CHK_ERR(RHS_.Reshape(NumRows_,NumVectors_));
206 IFPACK_CHK_ERR(LHS_.Reshape(NumRows_,NumVectors_));
207 // zero out vector elements
208 for (int i = 0 ; i < NumRows_ ; ++i)
209 for (int j = 0 ; j < NumVectors_ ; ++j) {
210 LHS_(i,j) = 0.0;
211 RHS_(i,j) = 0.0;
212 }
213 if (NumRows_!=0)
214 {
215 IFPACK_CHK_ERR(Solver_.SetVectors(LHS_,RHS_));
216 }
217 return(0);
218 }
