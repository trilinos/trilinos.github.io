74 int Ifpack_DiagPreconditioner::ApplyInverse(const Epetra_MultiVector& X, Epetra_MultiVector& Y) const
75{
76 if (X.NumVectors() != Y.NumVectors())
77 IFPACK_CHK_ERR(-1);
78
79 for (int v = 0; v < X.NumVectors(); ++v)
80 for (int i = 0; i < X.MyLength(); ++i)
81 Y[v][i] = diag_[i] * X[v][i];
83
84 return(0);
85}
