198 virtual int SetNumVectors(const int NumVectors_in)
199 {
200 if (NumVectors_ == NumVectors_in)
201 return(0);
202
203 NumVectors_ = NumVectors_in;
204 IFPACK_CHK_ERR(RHS_.Reshape(NumRows_,NumVectors_));
205 IFPACK_CHK_ERR(LHS_.Reshape(NumRows_,NumVectors_));
206 // zero out vector elements
207 for (int i = 0 ; i < NumRows_ ; ++i)
208 for (int j = 0 ; j < NumVectors_ ; ++j) {
209 LHS_(i,j) = 0.0;
210 RHS_(i,j) = 0.0;
211 }
212 if (NumRows_!=0)
213 {
214 IFPACK_CHK_ERR(Solver_.SetVectors(LHS_,RHS_));
215 }
216 return(0);
217 }
