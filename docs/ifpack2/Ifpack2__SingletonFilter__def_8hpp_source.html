22template <class MatrixType>
23SingletonFilter<MatrixType>::SingletonFilter(const Teuchos::RCP<const Tpetra::RowMatrix<Scalar, LocalOrdinal, GlobalOrdinal, Node> >& Matrix)
24 : A_(Matrix)
25 , NumSingletons_(0)
26 , NumRows_(0)
27 , NumNonzeros_(0)
28 , MaxNumEntries_(0)
29 , MaxNumEntriesA_(0) {
30 // use this filter only on serial matrices
31 if (A_->getComm()->getSize() != 1 || A_->getLocalNumRows() != A_->getGlobalNumRows()) {
32 throw std::runtime_error("Ifpack2::SingeltonFilter can be used with Comm().getSize() == 1 only. This class is a tool for Ifpack2_AdditiveSchwarz, and it is not meant to be used otherwise.");
33 }
34
35 // Number of rows in A
36 size_t NumRowsA_ = A_->getLocalNumRows();
37
38 // tentative value for MaxNumEntries. This is the number of
39 // nonzeros in the local matrix
40 MaxNumEntriesA_ = A_->getLocalMaxNumRowEntries();
41
42 // ExtractMyRowCopy() will use these vectors
43 Kokkos::resize(Indices_, MaxNumEntriesA_);
44 Kokkos::resize(Values_, MaxNumEntriesA_);
45
46 // Initialize reordering vector to -1
47 Reorder_.resize(NumRowsA_);
48 Reorder_.assign(Reorder_.size(), -1);
49
50 // first check how may singletons I do have
51 NumRows_ = 0;
52 for (size_t i = 0; i < NumRowsA_; ++i) {
53 size_t Nnz;
54 A_->getLocalRowCopy(i, Indices_, Values_, Nnz);
55 if (Nnz != 1) {
56 Reorder_[i] = NumRows_++;
57 } else {
58 NumSingletons_++;
59 }
60 }
61
62 // build the inverse reordering
63 InvReorder_.resize(NumRows_);
64 for (size_t i = 0; i < NumRowsA_; ++i) {
65 if (Reorder_[i] < 0)
66 continue;
67 InvReorder_[Reorder_[i]] = i;
68 }
69 NumEntries_.resize(NumRows_);
70 SingletonIndex_.resize(NumSingletons_);
71
72 // now compute the nonzeros per row
73 size_t count = 0;
74 for (size_t i = 0; i < NumRowsA_; ++i) {
75 size_t Nnz;
76 A_->getLocalRowCopy(i, Indices_, Values_, Nnz);
77 LocalOrdinal ii = Reorder_[i];
78 if (ii >= 0) {
79 NumEntries_[ii] = Nnz;
80 NumNonzeros_ += Nnz;
81 if (Nnz > MaxNumEntries_)
82 MaxNumEntries_ = Nnz;
83 } else {
84 SingletonIndex_[count] = i;
85 count++;
86 }
87 }
88
89 // Build the reduced map. This map should be serial
90 ReducedMap_ = Teuchos::rcp(new Tpetra::Map<LocalOrdinal, GlobalOrdinal, Node>(NumRows_, 0, A_->getComm()));
91
92 // and finish up with the diagonal entry
93 Diagonal_ = Teuchos::rcp(new Tpetra::Vector<Scalar, LocalOrdinal, GlobalOrdinal, Node>(ReducedMap_));
94
95 Tpetra::Vector<Scalar, LocalOrdinal, GlobalOrdinal, Node> DiagonalA(A_->getRowMap());
96 A_->getLocalDiagCopy(DiagonalA);
97 const Teuchos::ArrayRCP<const Scalar>& DiagonalAview = DiagonalA.get1dView();
98 for (size_t i = 0; i < NumRows_; ++i) {
99 LocalOrdinal ii = InvReorder_[i];
100 Diagonal_->replaceLocalValue((LocalOrdinal)i, DiagonalAview[ii]);
101 }
102}

242template <class MatrixType>
243void SingletonFilter<MatrixType>::
244 getLocalRowCopy(LocalOrdinal LocalRow,
245 nonconst_local_inds_host_view_type& Indices,
246 nonconst_values_host_view_type& Values,
247 size_t& NumEntries) const {
248 TEUCHOS_TEST_FOR_EXCEPTION((LocalRow < 0 || (size_t)LocalRow >= NumRows_ || (size_t)Indices.size() < NumEntries_[LocalRow]), std::runtime_error, "Ifpack2::SingletonFilter::getLocalRowCopy invalid row or array size.");
249
250 size_t Nnz;
251 LocalOrdinal ARow = InvReorder_[LocalRow];
252 A_->getLocalRowCopy(ARow, Indices_, Values_, Nnz);
253
254 // populate the user's vectors
255 NumEntries = 0;
256 for (size_t i = 0; i < Nnz; ++i) {
257 LocalOrdinal ii = Reorder_[Indices_[i]];
258 if (ii >= 0) {
259 Indices[NumEntries] = ii;
260 Values[NumEntries] = Values_[i];
261 NumEntries++;
262 }
263 }
264}

296template <class MatrixType>
297void SingletonFilter<MatrixType>::apply(const Tpetra::MultiVector<Scalar, LocalOrdinal, GlobalOrdinal, Node>& X,
298 Tpetra::MultiVector<Scalar, LocalOrdinal, GlobalOrdinal, Node>& Y,
299 Teuchos::ETransp mode,
300 Scalar /* alpha */,
301 Scalar /* beta */) const {
302 typedef Scalar DomainScalar;
303 typedef Scalar RangeScalar;
304
305 // Note: This isn't AztecOO compliant. But neither was Ifpack's version.
306
307 TEUCHOS_TEST_FOR_EXCEPTION(X.getNumVectors() != Y.getNumVectors(), std::runtime_error,
308 "Ifpack2::SingletonFilter::apply ERROR: X.getNumVectors() != Y.getNumVectors().");
309
310 RangeScalar zero = Teuchos::ScalarTraits<RangeScalar>::zero();
311 Teuchos::ArrayRCP<Teuchos::ArrayRCP<const DomainScalar> > x_ptr = X.get2dView();
312 Teuchos::ArrayRCP<Teuchos::ArrayRCP<RangeScalar> > y_ptr = Y.get2dViewNonConst();
313
314 Y.putScalar(zero);
315 size_t NumVectors = Y.getNumVectors();
316
317 for (size_t i = 0; i < NumRows_; ++i) {
318 size_t Nnz;
319 // Use this class's getrow to make the below code simpler
320 getLocalRowCopy(i, Indices_, Values_, Nnz);
321 if (mode == Teuchos::NO_TRANS) {
322 for (size_t j = 0; j < Nnz; ++j)
323 for (size_t k = 0; k < NumVectors; ++k)
324 y_ptr[k][i] += (RangeScalar)Values_[j] * (RangeScalar)x_ptr[k][Indices_[j]];
325 } else if (mode == Teuchos::TRANS) {
326 for (size_t j = 0; j < Nnz; ++j)
327 for (size_t k = 0; k < NumVectors; ++k)
328 y_ptr[k][Indices_[j]] += (RangeScalar)Values_[j] * (RangeScalar)x_ptr[k][i];
329 } else { // mode==Teuchos::CONJ_TRANS
330 for (size_t j = 0; j < Nnz; ++j)
331 for (size_t k = 0; k < NumVectors; ++k)
332 y_ptr[k][Indices_[j]] += Teuchos::ScalarTraits<RangeScalar>::conjugate((RangeScalar)Values_[j]) * (RangeScalar)x_ptr[k][i];
333 }
334 }
335}

353template <class MatrixType>
354template <class DomainScalar, class RangeScalar>
355void SingletonFilter<MatrixType>::SolveSingletonsTempl(const Tpetra::MultiVector<DomainScalar, LocalOrdinal, GlobalOrdinal, Node>& RHS,
356 Tpetra::MultiVector<RangeScalar, LocalOrdinal, GlobalOrdinal, Node>& LHS) {
357 Teuchos::ArrayRCP<Teuchos::ArrayRCP<const DomainScalar> > RHS_ptr = RHS.get2dView();
358 Teuchos::ArrayRCP<Teuchos::ArrayRCP<RangeScalar> > LHS_ptr = LHS.get2dViewNonConst();
359
360 for (size_t i = 0; i < NumSingletons_; ++i) {
361 LocalOrdinal ii = SingletonIndex_[i];
362 // get the diagonal value for the singleton
363 size_t Nnz;
364 A_->getLocalRowCopy(ii, Indices_, Values_, Nnz);
365 for (size_t j = 0; j < Nnz; ++j) {
366 if (Indices_[j] == ii) {
367 for (size_t k = 0; k < LHS.getNumVectors(); ++k)
368 LHS_ptr[k][ii] = (RangeScalar)RHS_ptr[k][ii] / (RangeScalar)Values_[j];
369 }
370 }
371 }
372}

381template <class MatrixType>
382template <class DomainScalar, class RangeScalar>
383void SingletonFilter<MatrixType>::CreateReducedRHSTempl(const Tpetra::MultiVector<DomainScalar, LocalOrdinal, GlobalOrdinal, Node>& LHS,
384 const Tpetra::MultiVector<RangeScalar, LocalOrdinal, GlobalOrdinal, Node>& RHS,
385 Tpetra::MultiVector<RangeScalar, LocalOrdinal, GlobalOrdinal, Node>& ReducedRHS) {
386 Teuchos::ArrayRCP<Teuchos::ArrayRCP<const RangeScalar> > RHS_ptr = RHS.get2dView();
387 Teuchos::ArrayRCP<Teuchos::ArrayRCP<const DomainScalar> > LHS_ptr = LHS.get2dView();
388 Teuchos::ArrayRCP<Teuchos::ArrayRCP<RangeScalar> > ReducedRHS_ptr = ReducedRHS.get2dViewNonConst();
389
390 size_t NumVectors = LHS.getNumVectors();
391
392 for (size_t i = 0; i < NumRows_; ++i)
393 for (size_t k = 0; k < NumVectors; ++k)
394 ReducedRHS_ptr[k][i] = RHS_ptr[k][InvReorder_[i]];
395
396 for (size_t i = 0; i < NumRows_; ++i) {
397 LocalOrdinal ii = InvReorder_[i];
398 size_t Nnz;
399 A_->getLocalRowCopy(ii, Indices_, Values_, Nnz);
400
401 for (size_t j = 0; j < Nnz; ++j) {
402 if (Reorder_[Indices_[j]] == -1) {
403 for (size_t k = 0; k < NumVectors; ++k)
404 ReducedRHS_ptr[k][i] -= (RangeScalar)Values_[j] * (RangeScalar)LHS_ptr[k][Indices_[j]];
405 }
406 }
407 }
408}
