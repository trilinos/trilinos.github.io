22template <class MatrixType>
23DiagonalFilter<MatrixType>::
24 DiagonalFilter(const Teuchos::RCP<const Tpetra::RowMatrix<Scalar, LocalOrdinal, GlobalOrdinal, Node> > &Matrix,
25 typename Teuchos::ScalarTraits<Scalar>::magnitudeType AbsoluteThreshold,
26 typename Teuchos::ScalarTraits<Scalar>::magnitudeType RelativeThreshold)
27 : A_(Matrix)
28 , AbsoluteThreshold_(AbsoluteThreshold)
29 , RelativeThreshold_(RelativeThreshold) {
30 pos_.resize(getLocalNumRows());
31 val_ = Teuchos::rcp(new Tpetra::Vector<Scalar, LocalOrdinal, GlobalOrdinal, Node>(A_->getRowMap()));
32
33 nonconst_local_inds_host_view_type Indices("Indices", getLocalMaxNumRowEntries());
34 nonconst_values_host_view_type Values("Values", getLocalMaxNumRowEntries());
35 size_t NumEntries;
36 magnitudeType mysign;
37
38 for (size_t MyRow = 0; MyRow < getLocalNumRows(); ++MyRow) {
39 pos_[MyRow] = -1;
40 A_->getLocalRowCopy(MyRow, Indices, Values, NumEntries);
41
42 for (size_t i = 0; i < NumEntries; ++i) {
43 if ((size_t)Indices[i] == MyRow) {
44 pos_[MyRow] = i;
45 if (Teuchos::ScalarTraits<Scalar>::real(Values[i]) < 0) {
46 mysign = -Teuchos::ScalarTraits<magnitudeType>::one();
47 } else {
48 mysign = Teuchos::ScalarTraits<magnitudeType>::one();
49 }
50 val_->replaceLocalValue(MyRow, Values[i] * (RelativeThreshold_ - 1) +
51 AbsoluteThreshold_ * mysign);
52 break;
53 }
54 }
55 }
56}

186template <class MatrixType>
187void DiagonalFilter<MatrixType>::
188 getGlobalRowCopy(GlobalOrdinal GlobalRow,
189 nonconst_global_inds_host_view_type &Indices,
190 nonconst_values_host_view_type &Values,
191 size_t &NumEntries) const
192
193{
194 Teuchos::ArrayRCP<const Scalar> myvals = val_->get1dView();
195 LocalOrdinal LocalRow = getRowMap()->getLocalElement(GlobalRow);
196
197 A_->getGlobalRowCopy(GlobalRow, Indices, Values, NumEntries);
198
199 if (pos_[LocalRow] != -1)
200 Values[pos_[LocalRow]] += myvals[LocalRow];
201}

248template <class MatrixType>
249void DiagonalFilter<MatrixType>::
250 apply(const Tpetra::MultiVector<Scalar, LocalOrdinal, GlobalOrdinal, Node> &X,
251 Tpetra::MultiVector<Scalar, LocalOrdinal, GlobalOrdinal, Node> &Y,
252 Teuchos::ETransp mode,
253 Scalar alpha,
254 Scalar beta) const {
255 // FIXME (mfh 27 Jul 2015) If this is a "diagonal filter," why do we
256 // need to apply the whole matrix???
257
258 Scalar one = Teuchos::ScalarTraits<Scalar>::one();
259 A_->apply(X, Y, mode, alpha, beta);
260 Y.elementWiseMultiply(one, *val_, X, one);
261}
