35 template<>
36 KOKKOS_FORCEINLINE_FUNCTION
37 double epsilon<double>() {
38 typedef union {
39 long long i64;
40 double d64;
41 } dbl_64;
42
43 dbl_64 s;
44 s.d64 = 1;
45 s.i64++;
46 return (s.i64 < 0 ? 1 - s.d64 : s.d64 - 1);
47 }
48
49 template<>
50 KOKKOS_FORCEINLINE_FUNCTION
51 float epsilon<float>() {
52 typedef union {
53 int i32;
54 float f32;
55 } flt_32;
56
57 flt_32 s;
58 s.f32 = 1;
59 s.i32++;
60 return (s.i32 < 0 ? 1 - s.f32 : s.f32 - 1);
61 }
