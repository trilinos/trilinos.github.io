127 auto output = Kokkos::subview( _outputValues, Kokkos::ALL(), pt);

129 Serial<opType>::getValues( output, input );
130 break;
131 }
132 default: {
133 INTREPID2_TEST_FOR_ABORT( opType != OPERATOR_VALUE &&
