74 // approximate spacing of source sites
75 double h_spacing = 0.05;
76 int n_neg1_to_1 = 2*(1/h_spacing) + 1; // always odd
77
78 // number of source coordinate sites that will fill a box of [-1,1]x[-1,1]x[-1,1] with a spacing approximately h
79 const int number_source_coords = std::pow(n_neg1_to_1, dimension);

81 // coordinates of source sites
82 Kokkos::View<double**, Kokkos::DefaultExecutionSpace> source_coords_device("source coordinates",

86 // coordinates of target sites
87 Kokkos::View<double**, Kokkos::DefaultExecutionSpace> target_coords_device ("target coordinates", number_target_coords, 3);
88 Kokkos::View<double**>::host_mirror_type target_coords = Kokkos::create_mirror_view(target_coords_device);
89
90 // tangent bundle for each target sites
91 Kokkos::View<double***, Kokkos::DefaultExecutionSpace> tangent_bundles_device ("tangent bundles", number_target_coords, 3, 3);
92 Kokkos::View<double***>::host_mirror_type tangent_bundles = Kokkos::create_mirror_view(tangent_bundles_device);

94 // fill source coordinates with a uniform grid
95 int source_index = 0;
96 double this_coord[3] = {0,0,0};
97 for (int i=-n_neg1_to_1/2; i<n_neg1_to_1/2+1; ++i) {
98 this_coord[0] = i*h_spacing;
99 for (int j=-n_neg1_to_1/2; j<n_neg1_to_1/2+1; ++j) {
100 this_coord[1] = j*h_spacing;
101 for (int k=-n_neg1_to_1/2; k<n_neg1_to_1/2+1; ++k) {
102 this_coord[2] = k*h_spacing;
103 if (dimension==3) {
104 source_coords(source_index,0) = this_coord[0];
105 source_coords(source_index,1) = this_coord[1];
106 source_coords(source_index,2) = this_coord[2];
107 source_index++;
108 }
109 }
110 if (dimension==2) {
111 source_coords(source_index,0) = this_coord[0];
112 source_coords(source_index,1) = this_coord[1];
113 source_coords(source_index,2) = 0;
114 source_index++;
115 }
116 }
117 if (dimension==1) {
118 source_coords(source_index,0) = this_coord[0];
119 source_coords(source_index,1) = 0;
120 source_coords(source_index,2) = 0;
121 source_index++;
122 }
123 }

125 // Generate target points - these are random permutation from available source points
126 // Note that this is assuming that the number of targets in this test will not exceed
127 // the number of source coords, which is 41^3 = 68921
128 // seed random number generator
129 std::mt19937 rng(50);
130 // generate random integers in [0..number_source_coords-1] (used to pick target sites)
131 std::uniform_int_distribution<int> gen_num_neighbours(0, source_index);
132 // fill target sites with random selections from source sites
133 for (int i=0; i<number_target_coords; ++i) {
134 const int source_site_to_copy = gen_num_neighbours(rng);
135 for (int j=0; j<3; j++) {
136 target_coords(i, j) = source_coords(source_site_to_copy, j);
137 }
138 if (constraint_name == "NEUMANN_GRAD_SCALAR") { // create bundles of normal vectors
139 if (dimension == 3) {
140 tangent_bundles(i, 0, 0) = 0.0;
141 tangent_bundles(i, 0, 1) = 0.0;
142 tangent_bundles(i, 0, 2) = 0.0;
143 tangent_bundles(i, 1, 0) = 0.0;
144 tangent_bundles(i, 1, 1) = 0.0;
145 tangent_bundles(i, 1, 2) = 0.0;
146 tangent_bundles(i, 2, 0) = 1.0/(sqrt(3.0));
147 tangent_bundles(i, 2, 1) = 1.0/(sqrt(3.0));
148 tangent_bundles(i, 2, 2) = 1.0/(sqrt(3.0));
149 }
150 if (dimension == 2) {
151 tangent_bundles(i, 0, 0) = 0.0;
152 tangent_bundles(i, 0, 1) = 0.0;
153 tangent_bundles(i, 1, 0) = 1.0/(sqrt(2.0));
154 tangent_bundles(i, 1, 1) = 1.0/(sqrt(2.0));
155 }
156 }

255 // pass in neighbor lists, source coordinates, target coordinates, and window sizes

257 // neighbor lists have the format:
258 // dimensions: (# number of target sites) X (# maximum number of neighbors for any given target + 1)

271 if (constraint_name == "NEUMANN_GRAD_SCALAR") {
272 scalar_basis_gmls.setTangentBundle(tangent_bundles_device);
273 vector_basis_gmls.setTangentBundle(tangent_bundles_device);

279 lro[1] = GradientOfScalarPointEvaluation;

290 scalar_basis_gmls.setWeightingParameter(2);
291 vector_basis_gmls.setWeightingParameter(2);

296 vector_basis_gmls.setQuadratureType("LINE");

311 // it is important to note that if you expect to use the data as a 1D view, then you should use double*
312 // however, if you know that the target operation will result in a 2D view (vector or matrix output),
313 // then you should template with double** as this is something that can not be infered from the input data

317 // The Evaluator class takes care of handling input data views as well as the output data views.

333

344 //! [Check That Solutions Are Correct]
345
346 // loop through the target sites
347 for (int i=0; i<number_target_coords; i++) {
348
349 // target site i's coordinate
350 double xval = target_coords(i,0);
351 double yval = (dimension>1) ? target_coords(i,1) : 0;
352 double zval = (dimension>2) ? target_coords(i,2) : 0;
353
354 // evaluation of various exact solutions
355 double actual_Divergence;
356 actual_Divergence = trueLaplacian(xval, yval, zval, order, dimension);
357 double actual_Gradient[3] = {0,0,0}; // initialized for 3, but only filled up to dimension
358 trueGradient(actual_Gradient, xval, yval, zval, order, dimension);
359
360 // calculate correction factor
361 double g = (dimension == 3) ? (1.0/sqrt(3.0))*(actual_Gradient[0] + actual_Gradient[1] + actual_Gradient[2])
362 : (1.0/sqrt(2.0))*(actual_Gradient[0] + actual_Gradient[1]);
363 // obtain number of neighbor for each target
364 // in order to exploit the index where the value for the Lagrange multiplier is stored
365 int num_neigh_i = neighbor_lists(i, 0);
366
367 // load divergence from output
368 double GMLS_Divergence_Scalar = output_divergence_scalar(i);
369 double GMLS_Divergence_Vector = output_divergence_vector(i);
370
371 // obtain adjusted value for divergence
372 if (constraint_name == "NEUMANN_GRAD_SCALAR") {
373 double b_i_scalar = scalar_basis_gmls.getSolutionSetHost()->getAlpha0TensorTo0Tensor(DivergenceOfVectorPointEvaluation, i, num_neigh_i);
374 GMLS_Divergence_Scalar = GMLS_Divergence_Scalar + b_i_scalar*g;
375
376 double b_i_vector = vector_basis_gmls.getSolutionSetHost()->getAlpha0TensorTo0Tensor(DivergenceOfVectorPointEvaluation, i, num_neigh_i);
377 GMLS_Divergence_Vector = GMLS_Divergence_Vector + b_i_vector*g;
378 }
379
380 // check divergence - scalar basis
381 if(std::abs(actual_Divergence - GMLS_Divergence_Scalar) > failure_tolerance) {
382 all_passed = false;
383 std::cout << i << " Failed Divergence on SCALAR basis by: " << std::abs(actual_Divergence - GMLS_Divergence_Scalar) << std::endl;
384 std::cout << i << " GMLS " << GMLS_Divergence_Scalar << " adjusted " << GMLS_Divergence_Scalar << " actual " << actual_Divergence << std::endl;
385 }
386
387 // check divergence - vector basis
388 if(std::abs(actual_Divergence - GMLS_Divergence_Vector) > failure_tolerance) {
389 all_passed = false;
390 std::cout << i << " Failed Divergence on VECTOR basis by: " << std::abs(actual_Divergence - GMLS_Divergence_Vector) << std::endl;
391 std::cout << i << " GMLS " << GMLS_Divergence_Vector << " adjusted " << GMLS_Divergence_Vector << " actual " << actual_Divergence << std::endl;
392 }
393
394 // load gradient from output
395 double Scalar_GMLS_GradX = (dimension>1) ? output_gradient_scalar(i,0) : 0;
396 double Scalar_GMLS_GradY = (dimension>1) ? output_gradient_scalar(i,1) : 0;
397 double Scalar_GMLS_GradZ = (dimension>2) ? output_gradient_scalar(i,2) : 0;
398
399 double Vector_GMLS_GradX = (dimension>1) ? output_gradient_vector(i,0) : 0;
400 double Vector_GMLS_GradY = (dimension>1) ? output_gradient_vector(i,1) : 0;
401 double Vector_GMLS_GradZ = (dimension>2) ? output_gradient_vector(i,2) : 0;
402
403 // Obtain adjusted value
404 if (constraint_name == "NEUMANN_GRAD_SCALAR") {
405 double bx_i_scalar = scalar_basis_gmls.getSolutionSetHost()->getAlpha0TensorTo1Tensor(GradientOfScalarPointEvaluation, i, 0, num_neigh_i);
406 double by_i_scalar = scalar_basis_gmls.getSolutionSetHost()->getAlpha0TensorTo1Tensor(GradientOfScalarPointEvaluation, i, 1, num_neigh_i);
407 double bz_i_scalar = scalar_basis_gmls.getSolutionSetHost()->getAlpha0TensorTo1Tensor(GradientOfScalarPointEvaluation, i, 2, num_neigh_i);
408 Scalar_GMLS_GradX = Scalar_GMLS_GradX + bx_i_scalar*g;
409 Scalar_GMLS_GradY = Scalar_GMLS_GradY + by_i_scalar*g;
410 Scalar_GMLS_GradZ = Scalar_GMLS_GradZ + bz_i_scalar*g;
411
412 double bx_i_vector = vector_basis_gmls.getSolutionSetHost()->getAlpha0TensorTo1Tensor(GradientOfScalarPointEvaluation, i, 0, num_neigh_i);
413 double by_i_vector = vector_basis_gmls.getSolutionSetHost()->getAlpha0TensorTo1Tensor(GradientOfScalarPointEvaluation, i, 1, num_neigh_i);
414 double bz_i_vector = vector_basis_gmls.getSolutionSetHost()->getAlpha0TensorTo1Tensor(GradientOfScalarPointEvaluation, i, 2, num_neigh_i);
415 Vector_GMLS_GradX = Vector_GMLS_GradX + bx_i_vector*g;
416 Vector_GMLS_GradY = Vector_GMLS_GradY + by_i_vector*g;
417 Vector_GMLS_GradZ = Vector_GMLS_GradZ + bz_i_vector*g;
418 }
419
420 // check gradient - scalar gmls
421 if(std::abs(actual_Gradient[0] - Scalar_GMLS_GradX) > failure_tolerance) {
422 all_passed = false;
423 std::cout << i << " Failed Scalar GradX by: " << std::abs(actual_Gradient[0] - Scalar_GMLS_GradX) << std::endl;
424 if (dimension>1) {
425 if(std::abs(actual_Gradient[1] - Scalar_GMLS_GradY) > failure_tolerance) {
426 all_passed = false;
427 std::cout << i << " Failed Scalar GradY by: " << std::abs(actual_Gradient[1] - Scalar_GMLS_GradY) << std::endl;
428 }
429 }
430 if (dimension>2) {
431 if(std::abs(actual_Gradient[2] - Scalar_GMLS_GradZ) > failure_tolerance) {
432 all_passed = false;
433 std::cout << i << " Failed Scalar GradZ by: " << std::abs(actual_Gradient[2] - Scalar_GMLS_GradZ) << std::endl;
434 }
435 }
436 }
437
438 // check gradient - vector gmls
439 if(std::abs(actual_Gradient[0] - Vector_GMLS_GradX) > failure_tolerance) {
440 all_passed = false;
441 std::cout << i << " Failed Vector GradX by: " << std::abs(actual_Gradient[0] - Vector_GMLS_GradX) << std::endl;
442 if (dimension>1) {
443 if(std::abs(actual_Gradient[1] - Vector_GMLS_GradY) > failure_tolerance) {
444 all_passed = false;
445 std::cout << i << " Failed Vector GradY by: " << std::abs(actual_Gradient[1] - Vector_GMLS_GradY) << std::endl;
446 }
447 }
448 if (dimension>2) {
449 if(std::abs(actual_Gradient[2] - Vector_GMLS_GradZ) > failure_tolerance) {
450 all_passed = false;
451 std::cout << i << " Failed Vector GradZ by: " << std::abs(actual_Gradient[2] - Vector_GMLS_GradZ) << std::endl;
452 }
453 }
454 }
455 }
