254 const Teuchos::RCP<const Map<LocalOrdinal, GlobalOrdinal, Node> >& tgt) {

260 "Tpetra::createExport: Neither source nor target map may be null.");
261#endif // HAVE_TPETRA_DEBUG
262 using export_type = Export<LocalOrdinal, GlobalOrdinal, Node>;
263 return Teuchos::rcp(new export_type(src, tgt));
