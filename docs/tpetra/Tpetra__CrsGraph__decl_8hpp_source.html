1536 pack_functor(DestViewType dest_,
1537 const SrcViewType src_,
1538 DestOffsetViewType dest_offset_,

1546 void operator()(size_t row) const {
1547 ScalarIndx i = src_offset(row);
1548 ScalarIndx j = dest_offset(row);

1550 for (; j < k; j++, i++) {
1551 dest(j) = src(i);
1552 }
1553 }
