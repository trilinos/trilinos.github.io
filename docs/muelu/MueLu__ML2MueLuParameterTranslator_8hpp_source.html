174 s.erase(s.begin(), std::find_if(s.begin(), s.end(), [](int c) { return !std::isspace(c); }));
175 return s;
176 }

178 // trim from end
179 static inline std::string& rtrim(std::string& s) {
180 s.erase(std::find_if(s.rbegin(), s.rend(), [](int c) { return !std::isspace(c); }).base(), s.end());
181 return s;
182 }
